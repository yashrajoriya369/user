import React from "react";

const logDisplay = (status) => {
  const element = document.getElementById("logDisplay");
  const isFullScreen = document.fullscreenElement;
  console.log(isFullScreen);
  if (status === "open") {
    element.requestFullscreen();
  } else {
    document.exitFullscreen();
  }
};

const Quiz = () => {
  return (
    <div id="logDisplay" style={{backgroundColor: "#FFF"}}>
      <button onClick={() => logDisplay("open")}>Open</button>
      <button onClick={() => logDisplay("close")}>Close</button>
    </div>
  );
};

export default Quiz;
